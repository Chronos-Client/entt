127template<typename Registry, typename Component>
128typename Registry::entity_type to_entity(const Registry &reg, const Component &instance) {
129 if(const auto *storage = reg.template storage<Component>(); storage) {
130 constexpr auto page_size = std::remove_const_t<std::remove_pointer_t<decltype(storage)>>::traits_type::page_size;
131 const typename Registry::common_type &base = *storage;
132 const auto *addr = std::addressof(instance);
133
134 for(auto it = base.rbegin(), last = base.rend(); it < last; it += page_size) {
135 if(const auto dist = (addr - std::addressof(storage->get(*it))); dist >= 0 && dist < static_cast<decltype(dist)>(page_size)) {
136 return *(it + dist);
137 }
138 }
139 }
