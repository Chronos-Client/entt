172 [[nodiscard]] auto sparse_ptr(const Entity entt) const {
173 const auto pos = static_cast<size_type>(traits_type::to_entity(entt));
174 const auto page = pos / traits_type::page_size;
175 return (page < sparse.size() && sparse[page]) ? (sparse[page] + fast_mod(pos, traits_type::page_size)) : nullptr;
176 }

178 [[nodiscard]] auto &sparse_ref(const Entity entt) const {
179 ENTT_ASSERT(sparse_ptr(entt), "Invalid element");
180 const auto pos = static_cast<size_type>(traits_type::to_entity(entt));
181 return sparse[pos / traits_type::page_size][fast_mod(pos, traits_type::page_size)];
182 }

184 [[nodiscard]] auto to_iterator(const Entity entt) const {
185 return --(end() - index(entt));

188 [[nodiscard]] auto &assure_at_least(const Entity entt) {
189 const auto pos = static_cast<size_type>(traits_type::to_entity(entt));
190 const auto page = pos / traits_type::page_size;
191
192 if(!(page < sparse.size())) {
193 sparse.resize(page + 1u, nullptr);
194 }
195
196 if(!sparse[page]) {
197 auto page_allocator{packed.get_allocator()};
198 sparse[page] = alloc_traits::allocate(page_allocator, traits_type::page_size);
199 std::uninitialized_fill(sparse[page], sparse[page] + traits_type::page_size, null);
200 }
201
202 auto &elem = sparse[page][fast_mod(pos, traits_type::page_size)];
203 ENTT_ASSERT(elem == null, "Slot not available");
204 return elem;
205 }
206
207 void release_sparse_pages() {
208 auto page_allocator{packed.get_allocator()};
209
210 for(auto &&page: sparse) {
211 if(page != nullptr) {
212 std::destroy(page, page + traits_type::page_size);
213 alloc_traits::deallocate(page_allocator, page, traits_type::page_size);
214 page = nullptr;
215 }
216 }

235 void swap_at(const std::size_t lhs, const std::size_t rhs) {
236 const auto entity = static_cast<typename traits_type::entity_type>(lhs);
237 const auto other = static_cast<typename traits_type::entity_type>(rhs);
238
239 sparse_ref(packed[lhs]) = traits_type::combine(other, traits_type::to_integral(packed[lhs]));
240 sparse_ref(packed[rhs]) = traits_type::combine(entity, traits_type::to_integral(packed[rhs]));
241
242 using std::swap;
243 swap(packed[lhs], packed[rhs]);
244 }

250 void swap_and_pop(const basic_iterator it) {
251 ENTT_ASSERT(mode == deletion_policy::swap_and_pop, "Deletion policy mismatched");
252 auto &self = sparse_ref(*it);
253 const auto entt = traits_type::to_entity(self);
254 sparse_ref(packed.back()) = traits_type::combine(entt, traits_type::to_integral(packed.back()));
255 packed[static_cast<size_type>(entt)] = packed.back();
256 // unnecessary but it helps to detect nasty bugs
257 ENTT_ASSERT((packed.back() = null, true), "");
258 // lazy self-assignment guard
259 self = null;
260 packed.pop_back();
261 }

267 void in_place_pop(const basic_iterator it) {
268 ENTT_ASSERT(mode == deletion_policy::in_place, "Deletion policy mismatched");
269 const auto entt = traits_type::to_entity(std::exchange(sparse_ref(*it), null));
270 packed[static_cast<size_type>(entt)] = std::exchange(free_list, traits_type::combine(entt, tombstone));

292 virtual void pop_all() {
293 if(const auto prev = std::exchange(free_list, tombstone); prev == null) {
294 for(auto first = begin(); !(first.index() < 0); ++first) {
295 sparse_ref(*first) = null;
296 }

314 virtual basic_iterator try_emplace(const Entity entt, const bool force_back, const void * = nullptr) {
315 ENTT_ASSERT(!contains(entt), "Set already contains entity");
316
317 if(auto &elem = assure_at_least(entt); free_list == null || force_back) {
318 packed.push_back(entt);
319 elem = traits_type::combine(static_cast<typename traits_type::entity_type>(packed.size() - 1u), traits_type::to_integral(entt));
320 return begin();
321 } else {
322 const auto pos = static_cast<size_type>(traits_type::to_entity(free_list));
323 elem = traits_type::combine(traits_type::to_integral(free_list), traits_type::to_integral(entt));
324 free_list = std::exchange(packed[pos], entt);
325 return --(end() - pos);
326 }
327 }

796 size_type remove(It first, It last) {
797 size_type count{};
798

824 void compact() {
825 size_type from = packed.size();
826 for(; from && packed[from - 1u] == tombstone; --from) {}
827
828 for(auto *it = &free_list; *it != null && from; it = std::addressof(packed[traits_type::to_entity(*it)])) {
829 if(const size_type to = traits_type::to_entity(*it); to < from) {
830 --from;
831 swap_or_move(from, to);
832
833 packed[to] = std::exchange(packed[from], tombstone);
834 const auto entity = static_cast<typename traits_type::entity_type>(to);
835 sparse_ref(packed[to]) = traits_type::combine(entity, traits_type::to_integral(packed[to]));
836
837 *it = traits_type::combine(static_cast<typename traits_type::entity_type>(from), tombstone);
838 for(; from && packed[from - 1u] == tombstone; --from) {}
839 }
840 }
841
842 free_list = tombstone;
843 packed.resize(from);
844 }

898 template<typename Compare, typename Sort = std_sort, typename... Args>
899 void sort_n(const size_type length, Compare compare, Sort algo = Sort{}, Args &&...args) {
900 ENTT_ASSERT(!(length > packed.size()), "Length exceeds the number of elements");
901 ENTT_ASSERT(free_list == null, "Partial sorting with tombstones is not supported");
902
903 algo(packed.rend() - length, packed.rend(), std::move(compare), std::forward<Args>(args)...);
904
905 for(size_type pos{}; pos < length; ++pos) {
906 auto curr = pos;
907 auto next = index(packed[curr]);
908
909 while(curr != next) {
910 const auto idx = index(packed[next]);
911 const auto entt = packed[curr];
912
913 swap_or_move(next, idx);
914 const auto entity = static_cast<typename traits_type::entity_type>(curr);
915 sparse_ref(entt) = traits_type::combine(entity, traits_type::to_integral(packed[curr]));
916 curr = std::exchange(next, idx);
917 }
918 }
919 }
