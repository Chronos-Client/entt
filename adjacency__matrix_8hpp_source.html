299 std::pair<edge_iterator, bool> insert(const vertex_type lhs, const vertex_type rhs) {
300 const auto pos = lhs * vert + rhs;
301
302 if constexpr(std::is_same_v<graph_category, undirected_tag>) {
303 const auto rev = rhs * vert + lhs;
304 ENTT_ASSERT(matrix[pos] == matrix[rev], "Something went really wrong");
305 matrix[rev] = 1u;
306 }
307
308 const auto inserted = !std::exchange(matrix[pos], 1u);
309 return {edge_iterator{matrix.cbegin(), vert, pos, matrix.size(), 1u}, inserted};
310 }

324 matrix[rev] = 0u;
