250 template<typename Type>
251 [[nodiscard]] auto &assure([[maybe_unused]] const id_type id = type_hash<Type>::value()) {
252 if constexpr(std::is_same_v<Type, entity_type>) {
253 return entities;
254 } else {
255 static_assert(std::is_same_v<Type, std::decay_t<Type>>, "Non-decayed types not allowed");
256 auto &cpool = pools[id];
257
258 if(!cpool) {
259 using storage_type = storage_for_type<Type>;
260 using alloc_type = typename storage_type::allocator_type;
261
262 if constexpr(std::is_same_v<Type, void> && !std::is_constructible_v<alloc_type, allocator_type>) {
263 // std::allocator<void> has no cross constructors (waiting for C++20)
264 cpool = std::allocate_shared<storage_type>(get_allocator(), alloc_type{});
265 } else {
266 cpool = std::allocate_shared<storage_type>(get_allocator(), get_allocator());
267 }
268
269 cpool->bind(forward_as_any(*this));
270 }
271
272 ENTT_ASSERT(cpool->type() == type_id<Type>(), "Unexpected type");
273 return static_cast<storage_for_type<Type> &>(*cpool);
274 }
275 }

846 template<typename Type, typename... Other, typename It>
847 size_type remove(It first, It last) {
848 size_type count{};
849
850 if constexpr(std::is_same_v<It, typename common_type::iterator>) {
851 common_type *cpools[sizeof...(Other) + 1u]{&assure<Type>(), &assure<Other>()...};
852
853 for(size_type pos{}, len = sizeof...(Other) + 1u; pos < len; ++pos) {
854 if constexpr(sizeof...(Other) != 0u) {
855 if(cpools[pos]->data() == first.data()) {
856 std::swap(cpools[pos], cpools[sizeof...(Other)]);
857 }
858 }
859
860 count += cpools[pos]->remove(first, last);
861 }
862 } else {
863 for(auto cpools = std::forward_as_tuple(assure<Type>(), assure<Other>()...); first != last; ++first) {
864 count += std::apply([entt = *first](auto &...curr) { return (curr.remove(entt) + ... + 0u); }, cpools);
865 }
866 }
867
868 return count;
869 }

1312 } else {
1313 cpool.sort(std::move(compare), std::move(algo), std::forward<Args>(args)...);
1314 }
