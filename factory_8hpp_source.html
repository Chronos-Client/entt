42inline meta_func_node &meta_extend(internal::meta_type_node &parent, const id_type id, meta_func_node node) {
43 if(auto it = parent.details->func.find(id); it != parent.details->func.end()) {
44 for(auto *curr = &it->second; curr; curr = curr->next.get()) {
45 if(curr->invoke == node.invoke) {
46 node.next = std::move(curr->next);
47 *curr = std::move(node);
48 return *curr;

439 template<typename... Value>
440 meta_factory prop(id_type id, [[maybe_unused]] Value &&...value) {
441 ENTT_ASSERT(bucket != nullptr, "Meta object does not support properties");
442
443 if constexpr(sizeof...(Value) == 0u) {
