265 template<typename Other>
266 [[nodiscard]] auto insert_or_do_nothing(Other &&value) {
267 const auto index = value_to_bucket(value);
268
269 if(auto it = constrained_find(value, index); it != end()) {
270 return std::make_pair(it, false);
271 }
272
273 packed.first().emplace_back(sparse.first()[index], std::forward<Other>(value));
274 sparse.first()[index] = packed.first().size() - 1u;
275 rehash_if_required();
276
277 return std::make_pair(--end(), true);

280 void move_and_pop(const std::size_t pos) {
281 if(const auto last = size() - 1u; pos != last) {
282 size_type *curr = sparse.first().data() + value_to_bucket(packed.first().back().second);
283 packed.first()[pos] = std::move(packed.first().back());
284 for(; *curr != last; curr = &packed.first()[*curr].first) {}
285 *curr = pos;
286 }
287
288 packed.first().pop_back();
289 }

291 void rehash_if_required() {
292 if(size() > (bucket_count() * max_load_factor())) {
293 rehash(bucket_count() * 2u);

523 template<typename... Args>
524 std::pair<iterator, bool> emplace(Args &&...args) {
525 if constexpr(((sizeof...(Args) == 1u) && ... && std::is_same_v<std::decay_t<Args>, value_type>)) {
526 return insert_or_do_nothing(std::forward<Args>(args)...);
527 } else {
528 auto &node = packed.first().emplace_back(std::piecewise_construct, std::make_tuple(packed.first().size()), std::forward_as_tuple(std::forward<Args>(args)...));
529 const auto index = value_to_bucket(node.second);
530
531 if(auto it = constrained_find(node.second, index); it != end()) {
532 packed.first().pop_back();
533 return std::make_pair(it, false);
534 }
535
536 std::swap(node.first, sparse.first()[index]);
537 rehash_if_required();
538
539 return std::make_pair(--end(), true);

836 void rehash(const size_type cnt) {
837 auto value = cnt > minimum_capacity ? cnt : minimum_capacity;
838 const auto cap = static_cast<size_type>(size() / max_load_factor());
839 value = value > cap ? value : cap;
840
841 if(const auto sz = next_power_of_two(value); sz != bucket_count()) {
842 sparse.first().resize(sz);
843
844 for(auto &&elem: sparse.first()) {
845 elem = std::numeric_limits<size_type>::max();
846 }
847
848 for(size_type pos{}, last = size(); pos < last; ++pos) {
849 const auto index = value_to_bucket(packed.first()[pos].second);
850 packed.first()[pos].first = std::exchange(sparse.first()[index], pos);
851 }
852 }
