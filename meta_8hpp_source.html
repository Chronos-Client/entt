159 template<typename Type>
160 static void basic_vtable([[maybe_unused]] const operation op, [[maybe_unused]] const any &value, [[maybe_unused]] void *other) {
161 static_assert(std::is_same_v<std::remove_cv_t<std::remove_reference_t<Type>>, Type>, "Invalid type");
162
163 if constexpr(!std::is_void_v<Type>) {
164 switch(op) {
165 case operation::deref:
166 if constexpr(is_meta_pointer_like_v<Type>) {
167 if constexpr(std::is_function_v<typename std::pointer_traits<Type>::element_type>) {
168 static_cast<meta_any *>(other)->emplace<Type>(any_cast<Type>(value));
169 } else if constexpr(!std::is_same_v<std::remove_const_t<typename std::pointer_traits<Type>::element_type>, void>) {
170 using in_place_type = decltype(adl_meta_pointer_like<Type>::dereference(any_cast<const Type &>(value)));
171
172 if constexpr(std::is_constructible_v<bool, Type>) {
173 if(const auto &pointer_like = any_cast<const Type &>(value); pointer_like) {
174 static_cast<meta_any *>(other)->emplace<in_place_type>(adl_meta_pointer_like<Type>::dereference(pointer_like));
175 }
176 } else {
177 static_cast<meta_any *>(other)->emplace<in_place_type>(adl_meta_pointer_like<Type>::dereference(any_cast<const Type &>(value)));
178 }
179 }
180 }
181 break;
182 case operation::seq:
183 if constexpr(is_complete_v<meta_sequence_container_traits<Type>>) {
184 static_cast<meta_sequence_container *>(other)->rebind<Type>(std::move(const_cast<any &>(value)));
185 }
186 break;
187 case operation::assoc:
188 if constexpr(is_complete_v<meta_associative_container_traits<Type>>) {
189 static_cast<meta_associative_container *>(other)->rebind<Type>(std::move(const_cast<any &>(value)));
190 }
191 break;
192 }
193 }
194 }

1079 [[nodiscard]] auto lookup(meta_any *const args, const typename internal::meta_type_node::size_type sz, [[maybe_unused]] bool constness, Func next) const {
1080 decltype(next()) candidate = nullptr;
1081 size_type same{};
1082 bool ambiguous{};
1083
1084 for(auto curr = next(); curr; curr = next()) {
1085 if constexpr(std::is_same_v<std::decay_t<decltype(*curr)>, internal::meta_func_node>) {
1086 if(constness && !static_cast<bool>(curr->traits & internal::meta_traits::is_const)) {
1087 continue;
1088 }
1089 }
1090
1091 if(curr->arity == sz) {
1092 size_type match{};
1093 size_type pos{};
1094
1095 for(; pos < sz && args[pos]; ++pos) {
1096 const auto other = curr->arg(*ctx, pos);
1097 const auto type = args[pos].type();
1098
1099 if(const auto &info = other.info(); info == type.info()) {
1100 ++match;
1101 } else if(!((type.node.details && (type.node.details->base.contains(info.hash()) || type.node.details->conv.contains(info.hash()))) || (type.node.conversion_helper && other.node.conversion_helper))) {
1102 break;
1103 }
1104 }
1105
1106 if(pos == sz) {
1107 if(!candidate || match > same) {
1108 candidate = curr;
1109 same = match;
1110 ambiguous = false;
1111 } else if(match == same) {
1112 if constexpr(std::is_same_v<std::decay_t<decltype(*curr)>, internal::meta_func_node>) {
1113 if(static_cast<bool>(curr->traits & internal::meta_traits::is_const) != static_cast<bool>(candidate->traits & internal::meta_traits::is_const)) {
1114 candidate = static_cast<bool>(candidate->traits & internal::meta_traits::is_const) ? curr : candidate;
1115 ambiguous = false;
1116 continue;
1117 }
1118 }
1119
1120 ambiguous = true;
1121 }
1122 }
1123 }
1124 }
1125
1126 return ambiguous ? nullptr : candidate;
1127 }
