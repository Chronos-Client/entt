35template<typename Type, typename Allocator>
36class dispatcher_handler final: public basic_dispatcher_handler {
37 static_assert(std::is_same_v<Type, std::decay_t<Type>>, "Invalid type");
38
39 using alloc_traits = std::allocator_traits<Allocator>;
40 using signal_type = sigh<void(Type &), Allocator>;
41 using container_type = std::vector<Type, typename alloc_traits::template rebind_alloc<Type>>;
42
43public:
44 using allocator_type = Allocator;
45
46 dispatcher_handler(const allocator_type &allocator)
47 : signal{allocator},
48 events{allocator} {}
49
50 void publish() override {
51 const auto length = events.size();
52
53 for(std::size_t pos{}; pos < length; ++pos) {
54 signal.publish(events[pos]);
55 }
56
57 events.erase(events.cbegin(), events.cbegin() + length);
58 }
59
60 void disconnect(void *instance) override {
61 bucket().disconnect(instance);
62 }
63
64 void clear() noexcept override {
65 events.clear();
66 }
67
68 [[nodiscard]] auto bucket() noexcept {
69 return typename signal_type::sink_type{signal};
70 }
71
72 void trigger(Type event) {
73 signal.publish(event);
74 }
75
76 template<typename... Args>
77 void enqueue(Args &&...args) {
78 if constexpr(std::is_aggregate_v<Type> && (sizeof...(Args) != 0u || !std::is_default_constructible_v<Type>)) {
79 events.push_back(Type{std::forward<Args>(args)...});
80 } else {
81 events.emplace_back(std::forward<Args>(args)...);
82 }
83 }
84
85 std::size_t size() const noexcept override {
86 return events.size();
87 }
88
89private:
90 signal_type signal;
91 container_type events;
92};
